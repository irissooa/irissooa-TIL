## BOJ_17413_단어뒤집기2

```python
words = input()
# print(len(words))
idx =0
q = []
ans = []

while True:
    if idx >= len(words):
        ans = ans + q[::-1]
        break
    #태그를 만나면 닫을때까지 q에담고 q를 그대로 ans에 담아줌
    if words[idx] == '<':
        if q:
            ans = ans + q[::-1]
            q= []
        while True:
            q.append(words[idx])
            if words[idx] == '>':
                ans += q
                q = []
                idx+=1
                break
            idx += 1
    #문자열
    else:
        if q == [' '] or words[idx] == ' ':
            ans = ans + q[::-1]
            q = []

        q.append(words[idx])
        idx+=1
print(''.join(ans))
```



- 다른사람 코드..이사람 대박

```python
#a = <ab cd>ef gh<ij kl>
a=input()
#a의 input값 중 '>'를 '<'로 바꾸고, '<'를 기준으로 나눠서 리스트에 담기
b=a.replace('>','<').split('<')
#b = ['', 'ab cd', 'ef gh', 'ij kl', '']
s=""

for i in range(len(b)):
  #i가 홀수면 '<>'문자열
  if i%2:
      s+='<'+b[i]+'>'
  else:
    c=b[i].split()
    s+=' '.join([d[::-1] for d in c])
print(s)
```

```python
S = input()
tmp = ''
ans = ''
isTag = False
 
for i in S:
    if i == ' ':
        ans += tmp[::-1] + i
        tmp = ''
    elif i == '<':
        isTag = True
        ans += tmp[::-1] + i
        tmp = ''
    elif i == '>':
        isTag = False
        ans += i
    elif isTag:
         ans += i
    else:
        tmp += i
 
ans += tmp[::-1]
 
print(ans)
```

```python
S = input()
 
tmp, ans, ck = '', '', False
 
for i in S:
    if i == ' ':
        if not ck:
            ans += tmp[::-1] + i
            tmp = ''
        else: ans += i
    elif i == '<':
        ck = True
        ans += tmp[::-1] + i
        tmp = ''
    elif i == '>':
        ck = False
        ans += i
    else:
        if ck: ans += i
        else: tmp += i
 
ans += tmp[::-1]
print(ans)
```
